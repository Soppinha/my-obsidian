## ex02

```c
#include <stdio.h>

void ft_swap(int *a, int *b);

void ft_swap(int *a, int *b)
{
	int aux;
	
	aux = *a;
	*a = *b;
	*b = aux;
}

int main()
{
    int x = 5;
    int y = 10;  

    printf("Antes da troca: x = %d, y = %d\n", x, y);

    ft_swap(&x, &y);

    printf("Depois da troca: x = %d, y = %d\n", x, y);

    return 0;
}
```
